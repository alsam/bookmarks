      + tl;dr  
      ```c++
      for (const auto &face : faces(poly))
      {
        auto begin = face->halfedge();
        auto edge = begin;
        // Each vertex
        std::vector<Point_3> tri;
        do
        {
          Point_3 p = edge->vertex()->point();
          tri.push_back(std::move(p));
          edge = edge->next();
        } while (edge != begin);
        ASSERT_TRUE(tri.size() == 3);
      }
      ```
